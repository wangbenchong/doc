# 附C#控制台代码：


```c#
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;

namespace ChineseCharManager
{
    class Program
    {
        static void Main(string[] args)
        {
            
            Dictionary<char, int> chineseCharCount = new Dictionary<char, int>();
            while(true)
            {
                Console.WriteLine("请输入待分析文件路径并回车(务必是utf8编码的纯文本文件)，直接回车则跳转到生僻字排除");
                string filePath = Console.ReadLine().Replace("\"", ""); ;
                if(filePath.Length < 2)
                {
                    Export(chineseCharCount);
                    break;
                }
                if(!File.Exists(filePath))
                {
                    Console.WriteLine("错误! 文件不存在，请检查路径" + filePath);
                    continue;
                }
                // 读取文件内容  
                string content = File.ReadAllText(filePath, Encoding.UTF8);

                // 遍历文件中的每个字符  
                foreach (char c in content)
                {
                    // 检查是否为中文字符  
                    if (IsChineseChar(c))
                    {
                        // 如果字典中已存在该字符，则增加其计数  
                        if (chineseCharCount.ContainsKey(c))
                        {
                            chineseCharCount[c]++;
                        }
                        else
                        {
                            // 否则，添加到字典中，计数为1  
                            chineseCharCount.Add(c, 1);
                        }
                    }
                }

                Console.WriteLine("字典长度" + chineseCharCount.Count);
            }
        }

        // 辅助方法：检查一个字符是否为中文字符  
        static bool IsChineseChar(char c)
        {
            // 中文字符的Unicode编码范围大致为\u4e00到\u9fa5  
            return c >= '\u4e00' && c <= '\u9fa5';
        }
        static void Export(Dictionary<char, int> chineseCharCount)
        {
            Console.WriteLine("请输入待排除的生僻字文件(务必是utf8编码的纯文本文件)，直接回车则执行导出当前字数统计结果");
            string filePath = Console.ReadLine().Replace("\"","");
            if (filePath.Length >= 2 && File.Exists(filePath))
            {
                // 读取文件内容  
                string content = File.ReadAllText(filePath, Encoding.UTF8);
                // 遍历文件中的每个字符  
                foreach (char c in content)
                {
                    // 检查是否为中文字符  
                    if (IsChineseChar(c))
                    {
                        chineseCharCount.Remove(c);
                    }
                }
                Console.WriteLine("字典长度" + chineseCharCount.Count);
            }
            Console.WriteLine("开始执行文件导出...");
            string outputFilePath = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "中文字符数量统计.txt"); // 输出文件路径
            string outputFilePath2 = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "中文字符紧凑输出.txt"); // 输出文件路径
                                                                                                          // 写入到输出文件  
            var sortResult1 = from pair in chineseCharCount orderby pair.Value descending select pair;
            using (StreamWriter writer = new StreamWriter(outputFilePath, false, Encoding.UTF8))
            {
                foreach (var kvp in sortResult1)
                {
                    writer.WriteLine($"{kvp.Key} {kvp.Value}");
                }
            }
            using (StreamWriter writer = new StreamWriter(outputFilePath2, false, Encoding.UTF8))
            {
                foreach (var kvp in sortResult1)
                {
                    writer.Write(kvp.Key);
                }
            }

            Console.WriteLine("字典信息已导出到:\n" + outputFilePath + "\n以及:\n" + outputFilePath2);
            Console.ReadLine();
        }
    }
}

```
